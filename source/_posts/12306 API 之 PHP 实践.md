        $date = $date ?: date('Y-m-d');

        $ticket = $this->ticket($from, $to, $date, false);

        //dump($ticket);

        if (!isset($ticket['result'])) {
            return json(['status' => 404], 404);
        }

        foreach ($ticket['result'] as $key => $each) {
            $res = json_decode(file_get_contents("https://kyfw.12306.cn/otn/leftTicket/queryTicketPrice?train_no={$each['train_no']}&from_station_no={$each['from_station_no']}&to_station_no={$each['to_station_no']}&seat_types={$each['seat_types']}&train_date=$date",
                false,
                stream_context_create(['ssl' => ['verify_peer' => false,]])
            ), true);

            $price_list = [];
            foreach ($res['data'] as $k => $v) {
                if (array_key_exists($k, $seatType)) {
                    $price_list[$seatType[$k]] = $v;
                }
            }
            //dump($price_list);
            $ticket['result'][$key]['price_list'] = $price_list;
        }

        //dump($ticket);
        return json($ticket, $ticket['status']);

    }
